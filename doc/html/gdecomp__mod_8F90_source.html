00884    subroutine calcbsiz(npes,gsiz,bsiz,option,ierr)
00885 
00886    implicit none
00887 
00888    integer(i4), intent(in)    :: npes
00889    integer(i4), intent(in)    :: gsiz(:)
00890    integer(i4), intent(inout) :: bsiz(:)

00893 
00894    integer(i4) :: gs,bs
00895    integer(i4),allocatable :: nsiz(:),isiz(:)
00896    integer(i4) :: npes2
00897    integer(i4) :: bs1,bs2,bs3
00898    real(r8)    :: rbs1,rbs2,rbs3
00899    integer(i4) :: n,m,nx,ny,nz,n1,n2,n3
00900    real(r8)    :: ratio,ratio2
00901    integer(i4) :: nbsiz
00902    real(r8),parameter :: dbsizm = 0.7      ! 0.0 turns this off
00903    real(r8),parameter :: dbsizp = 1.3   ! big number turns this off (npes)

00906    character(len=*),parameter :: subname = 'calcbsiz'

00909

00911       if (trim(option) == 'ediv') then

00915       else
00916          write(6,*) trim(subname),' ERROR: option not valid ',trim(option)
00917          call piodie(__FILE__,__LINE__)
00918       endif
00919    endif
00920 
00921    found = .false.
00922    gs = size(gsiz)
00923    bs = size(bsiz)
00924    if (gs /= 3) then
00925       write(6,*) trim(subname),' ERROR: gs size must be 3 ',gs

00933

00935    do n = 1,3
00936       if (bsiz(n) /= 0) then
00937          call pad_div(m,gsiz(n),bsiz(n))
00938          if (mod(npes2,m) == 0) then
00939             nsiz(n) = m
00940             npes2 = npes2/m
00941             bs = bs - 1
00942          else
00943             write(6,*) trim(subname),' ERROR: bsiz not allowed ',n,gsiz(n),bsiz(n),m,npes,npes2

00950    isiz = 0
00951    do n = 1,3
00952       if (bsiz(n) == 0) then
00953          bs = bs + 1
00954          isiz(bs) = n
00955       endif
00956    enddo
00957    n1 = isiz(1)
00958    n2 = isiz(2)
00959    n3 = isiz(3)
00960 
00961    if (bs == 1) then
00962       nsiz(n1) = npes2
00963       if (check_ediv(ediv,gsiz(n1),nsiz(n1))) then

00971       nbsiz = 10.*gsiz(1)*gsiz(2)

00974             ny = npes2/nx
00975             if (check_ediv(ediv,gsiz(n1),nx) .and. check_ediv(ediv,gsiz(n2),ny)) then
00976                call pad_div(bs1,gsiz(n1),nx)
00977                call pad_div(bs2,gsiz(n2),ny)
00978                rbs1 = bs1
00979                rbs2 = bs2
00980 !               if (max(rbs1/rbs2,rbs2/rbs1) < ratio) then
00981                if ((bs1*bs2 < (dbsizm)*nbsiz) .or. &
00982                    (bs1*bs2 < (dbsizp)*nbsiz .and. max(rbs1/rbs2,rbs2/rbs1) < ratio)) then
00983                   ratio = max(rbs1/rbs2,rbs2/rbs1)
00984                   nbsiz = bs1*bs2
00985                   bsiz(n1) = bs1
00986                   bsiz(n2) = bs2
00987                   nsiz(n1) = nx
00988                   nsiz(n2) = ny
00989                   found = .true.
00990                endif
00991             endif
00992          endif
00993       enddo
00994    endif
00995 
00996    if (bs == 3) then
00997       ratio = 10.*gsiz(1)*gsiz(2)*gsiz(3)
00998       nbsiz = 10.*gsiz(1)*gsiz(2)*gsiz(3)
00999       do nx = 1,npes2
01000          if (mod(npes2,nx) == 0) then
01001             do ny = 1,npes2/nx
01002                if (mod(npes2/nx,ny) == 0) then
01003                   nz = npes2/(nx*ny)
01004                   if (check_ediv(ediv,gsiz(n1),nx) .and. check_ediv(ediv,gsiz(n2),ny) .and. check_ediv(ediv,gsiz(n3),nz)) then
01005                      call pad_div(bs1,gsiz(n1),nx)
01006                      call pad_div(bs2,gsiz(n2),ny)
01007                      call pad_div(bs3,gsiz(n3),nz)
01008                      rbs1 = bs1
01009                      rbs2 = bs2
01010                      rbs3 = bs3
01011                      ratio2 = max(rbs1/rbs2,rbs2/rbs1)
01012                      ratio2 = max(ratio2,rbs1/rbs3)
01013                      ratio2 = max(ratio2,rbs3/rbs1)
01014                      ratio2 = max(ratio2,rbs2/rbs3)
01015                      ratio2 = max(ratio2,rbs3/rbs2)
01016 !                     if (ratio2 < ratio) then
01017                      if ((bs1*bs2*bs3 < (dbsizm)*nbsiz) .or. &
01018                          (bs1*bs2*bs3 < (dbsizp)*nbsiz .and. ratio2 < ratio)) then
01019                         ratio = ratio2
01020                         nbsiz = bs1*bs2*bs3
01021                         bsiz(n1) = bs1
01022                         bsiz(n2) = bs2
01023                         bsiz(n3) = bs3
01024                         nsiz(n1) = nx
01025                         nsiz(n2) = ny
01026                         nsiz(n3) = nz
01027                         found = .true.
01028                      endif
01029                   endif
01030                endif
01031             enddo
01032          endif
01033       enddo
01034    endif
01035 
01036    if (found) then
01037 !      write(6,*) trim(subname),' found; gsiz=',gsiz,'npes=',npes,'nsiz=',nsiz,'bsiz=',bsiz
01038    else
01039       write(6,*) trim(subname),' ERROR: no decomp found gsiz=',gsiz,' npes=',npes
01040       call piodie(__FILE__,__LINE__)
01041    endif
01042 
01043    deallocate(nsiz,isiz)
01044 
01045    end subroutine calcbsiz

01050 
01051    implicit none
01052    integer(i4), intent(out) :: mout
01053    integer(i4), intent(in)  :: num
