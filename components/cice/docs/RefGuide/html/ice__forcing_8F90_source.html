01300          swvdf   ,  ! sw down, visible, diffuse (W/m^2)
01301          swidr   ,  ! sw down, near IR, direct  (W/m^2)
01302          swidf   ,  ! sw down, near IR, diffuse (W/m^2)

01306 !
01307       integer (kind=int_kind) :: 
01308          i, j
01309

01311          fcc, sstk, rtea, ptem, qlwm, precip_factor
01312

01314       do i = ilo, ihi
01315 
01316       !-----------------------------------------------------------------
01317       ! make sure interpolated values are physically realistic
01318       !-----------------------------------------------------------------
01319          cldf (i,j) = max(min(cldf(i,j),c1),c0)
01320          fsw  (i,j) = max(fsw(i,j),c0)
01321          fsnow(i,j) = max(fsnow(i,j),c0)
01322          rhoa (i,j) = max(rhoa(i,j),c0)
01323          Qa   (i,j) = max(Qa(i,j),c0)
01324 
01325       enddo                     ! i
01326       enddo                     ! j
01327 
01328       !-----------------------------------------------------------------
01329       ! calculations specific to datasets
01330       !-----------------------------------------------------------------
01331 
01332       if (trim(atm_data_type) == 'ncar') then
01333          do j = jlo, jhi
01334          do i = ilo, ihi
01335 
01336       !-----------------------------------------------------------------
01337       ! correct known biases in NCAR data (as in CCSM latm)
01338       !-----------------------------------------------------------------
01339 
01340             Qa (i,j) = Qa (i,j) * 0.94_dbl_kind
01341             fsw(i,j) = fsw(i,j) * 0.92_dbl_kind
01342 
01343       !-----------------------------------------------------------------
01344       ! compute downward longwave as in Parkinson and Washington (1979)
01345       ! (for now)
01346       !-----------------------------------------------------------------
01347 
01348             flw(i,j) = stefan_boltzmann*Tair(i,j)**4 &
01349                      * (c1 - 0.261_dbl_kind &
01350                       *exp(-7.77e-4_dbl_kind*(Tffresh - Tair(i,j))**2)) &
01351                      * (c1 + 0.275_dbl_kind*cldf(i,j))
01352 
01353       ! precip is in mm/month; converted to mks below
01354 
01355          enddo
01356          enddo
01357 
01358       elseif (trim(atm_data_type) == 'ecmwf') then
01359          do j = jlo, jhi
01360          do i = ilo, ihi
01361 
01362       !-----------------------------------------------------------------
01363       ! The following assumes that the input Qa is really dew point temp
01364       ! (deg K) and need to be converted to specific humidity (kg/kg).
01365       ! Cf. ice_atmo module.
01366       !-----------------------------------------------------------------
01367 
01368          Qa (i,j) = (qqqocn/rhoa(i,j)) * exp(-TTTocn/Qa(i,j))
01369          Qa (i,j) = max(Qa(i,j),c0)
01370 
01371       !-----------------------------------------------------------------
01372       ! compute downward longwave as in Parkinson and Washington (1979)
01373       ! (for now)
01374       !-----------------------------------------------------------------
01375 
01376          flw(i,j) = stefan_boltzmann*Tair(i,j)**4 &
01377                   * (c1 - 0.261_dbl_kind &
01378                    *exp(-7.77e-4_dbl_kind*(Tffresh - Tair(i,j))**2)) &
01379                   * (c1 + 0.275_dbl_kind*cldf(i,j))
01380 
01381       ! precip is in mm/month; converted to mks below
01382 
01383          enddo
01384          enddo

01390       !-----------------------------------------------------------------

01418          precip_factor = c1/secday
01419       elseif (trim(precip_units) == 'mm_per_sec' .or. &
01420               trim(precip_units) == 'mks') then

01422       endif

01425       do i = ilo, ihi
01426 
01427          zlvl(i,j) = c10
01428          potT(i,j) = Tair(i,j)
01429 
01430         ! divide shortwave into spectral bands
01431          swvdr(i,j) = fsw(i,j)*frcvdr        ! visible direct
01432          swvdf(i,j) = fsw(i,j)*frcvdf        ! visible diffuse

01434          swidf(i,j) = fsw(i,j)*frcidf        ! near IR diffuse
01435                  
01436         ! convert precipitation units to kg/m^2 s
01437          fsnow(i,j) = fsnow(i,j) * precip_factor
01438 
01439       enddo                     ! i
01440       enddo                     ! j
01441 
01442       ! determine whether precip is rain or snow
01443       ! HadGEM forcing provides separate snowfall and rainfall rather 
01444       ! than total precipitation
01445       if (trim(atm_data_type) /= 'hadgem') then
01446 
01447         do j = jlo, jhi
01448         do i = ilo, ihi
01449            frain(i,j) = c0                     
01450            if (Tair(i,j) >= Tffresh) then
01451                frain(i,j) = fsnow(i,j)
01452                fsnow(i,j) = c0
01453            endif
01454         enddo                     ! i
01455         enddo                     ! j
01456 
01457       endif
01458 
01459       if (calc_strair) then
01460 
01461         do j = jlo, jhi
01462         do i = ilo, ihi
01463 
01464             wind(i,j) = sqrt(uatm(i,j)**2 + vatm(i,j)**2)
01465 
01466       !-----------------------------------------------------------------
01467       ! Rotate zonal/meridional vectors to local coordinates.
01468       ! Velocity comes in on T grid, but is oriented geographically ---
01469       ! need to rotate to pop-grid FIRST using ANGLET
01470       ! then interpolate to the U-cell centers  (otherwise we
01471       ! interpolate across the pole).
01472       ! Use ANGLET which is on the T grid !
01473       ! Atmo variables are needed in T cell centers in subroutine 
01474       ! atmo_boundary_layer, and are interpolated to the U grid later as 
01475       ! necessary.
01476       !-----------------------------------------------------------------
01477            workx      = uatm(i,j) ! wind velocity, m/s
01478            worky      = vatm(i,j)
01479            uatm (i,j) = workx*cos(ANGLET(i,j)) & ! convert to POP grid
01480                       + worky*sin(ANGLET(i,j))   ! note uatm, vatm, wind
01481            vatm (i,j) = worky*cos(ANGLET(i,j)) & !  are on the T-grid here
01482                       - workx*sin(ANGLET(i,j))
01483 
01484         enddo                     ! i
01485         enddo                     ! j
01486 
01487       else  ! strax, stray, wind are read from files
01488 
01489         do j = jlo, jhi
01490         do i = ilo, ihi
01491 
01492            workx      = strax(i,j) ! wind stress
01493            worky      = stray(i,j)
01494            strax(i,j) = workx*cos(ANGLET(i,j)) & ! convert to POP grid
01495                       + worky*sin(ANGLET(i,j))   ! note strax, stray, wind
01496            stray(i,j) = worky*cos(ANGLET(i,j)) & !  are on the T-grid here
01497                       - workx*sin(ANGLET(i,j))
01498 
01499         enddo                     ! i
01500         enddo                     ! j
01501 
01502       endif                   ! calc_strair
01503 
01504       end subroutine prepare_forcing

02290       subroutine compute_shortwave(nx_block,  ny_block, &
02291                                    ilo, ihi, jlo, jhi, &
02292                                    TLON, TLAT, hm, Qa, cldf, fsw)
02293 
02294 !---!-------------------------------------------------------------------
02295 !---! AOMIP shortwave forcing
02296 !---!-------------------------------------------------------------------
02297 
02298       integer (kind=int_kind), intent(in) :: 
02299          nx_block, ny_block,  ! block dimensions
02300          ilo,ihi,jlo,jhi       ! beginning and end of physical domain
02301 
02302       real (kind=dbl_kind), dimension(nx_block,ny_block), intent(in) :: 
02303          TLON, TLAT     ,  ! longitude, latitude
02304          Qa             ,  ! specific humidity
02305          cldf           ,  ! cloud fraction
02306          hm                 ! land mask
02307 
02308       real (kind=dbl_kind), dimension(nx_block,ny_block),  
02309          intent(inout) :: 
02310          fsw                ! shortwave
02311

02313          hour_angle, 
02314          solar_time, 
02315          declin    , 
02316          cosZ      , 
02317          e, d      , 
02318          sw0       , 
02319          deg2rad   
02320 
02321       integer (kind=int_kind) :: 
02322          i, j

02328                    + c12*sin(p5*TLON(i,j))
02329         hour_angle = (c12 - solar_time)*pi/c12
02330         declin = 23.44_dbl_kind*cos((172._dbl_kind-yday) &
02331                  * c2*pi/c365)*deg2rad     ! use dayyr instead of c365???
02332         cosZ = sin(TLAT(i,j))*sin(declin) &
02333              + cos(TLAT(i,j))*cos(declin)*cos(hour_angle)
02334         cosZ = max(cosZ,c0)
02335         e = 1.e5*Qa(i,j)/(0.622_dbl_kind + 0.378_dbl_kind*Qa(i,j))
02336         d = (cosZ+2.7_dbl_kind)*e*1.e-5_dbl_kind+1.085_dbl_kind*cosZ+p1

03597       subroutine ocn_data_hadgem(dt)
03598 !
03599 ! !DESCRIPTION:
03600 !  Reads in HadGEM ocean forcing data as required from netCDF files
03601 !  Current options (selected by sst_data_type)
03602 !  hadgem_sst:          read in sst only 
03603 !  hadgem_sst_uvocn:    read in sst plus uocn and vocn  
03604 !        
03605 !
03606 ! !REVISION HISTORY:
03607 !
03608 ! authors: Ann Keen, Met Office
03609 !
03610 ! !USES:
03611 !
03612       use ice_domain, only: nblocks
03613       use ice_flux, only: sst, uocn, vocn
03614       use ice_grid, only: t2ugrid_vector, ANGLET
03615 !
03616 ! !INPUT/OUTPUT PARAMETERS:
03617 !
03618 !
03619 !EOP
03620 !
03621       real (kind=dbl_kind), intent(in) :: 
03622          dt      ! time step
03623  
03624      integer (kind=int_kind) :: 
03625           i, j        ,  ! horizontal indices
03626           n           ,  ! thickness category index
03627           iblk        ,  ! block index
03628           ixm,ixx,ixp ,  ! record numbers for neighboring months
03629           recnum      ,  ! record number
03630           maxrec      ,  ! maximum record number
03631           recslot     ,  ! spline slot for current record
03632           dataloc     ,  ! = 1 for data located in middle of time interval
03633                           ! = 2 for date located at end of time interval
03634           midmonth        ! middle day of month
03635 
03636       real (kind=dbl_kind), dimension(nx_block,ny_block,max_blocks) :: 
03637           sstdat              ! data value toward which SST is restored
03638 
03639       real (kind=dbl_kind) :: workx, worky
03640 
03641       logical (kind=log_kind) :: readm
03642 
03643       character (char_len) ::  
03644             fieldname           ! field name in netcdf file
03645 
03646       character (char_len_long) ::  
03647             filename            ! name of netCDF file
03648 
03649     !-------------------------------------------------------------------
03650     ! monthly data
03651     !
03652     ! Assume that monthly data values are located in the middle of the
03653     ! month.
03654     !-------------------------------------------------------------------
03655 
03656       midmonth = 15  ! data is given on 15th of every month
03657 !      midmonth = fix(p5 * real(daymo(month)))  ! exact middle
03658 
03659       ! Compute record numbers for surrounding months
03660       maxrec = 12
03661       ixm  = mod(month+maxrec-2,maxrec) + 1
03662       ixp  = mod(month,         maxrec) + 1
03663       if (mday >= midmonth) ixm = 99  ! other two points will be used
03664       if (mday <  midmonth) ixp = 99
03665 
03666       ! Determine whether interpolation will use values 1:2 or 2:3
03667       ! recslot = 2 means we use values 1:2, with the current value (2)
03668       !  in the second slot
03669       ! recslot = 1 means we use values 2:3, with the current value (2)
03670       !  in the first slot
03671       recslot = 1                             ! latter half of month
03672       if (mday < midmonth) recslot = 2        ! first half of month
03673 
03674       ! Find interpolation coefficients
03675       call interp_coeff_monthly (recslot)
03676 
03677       ! Read 2 monthly values
03678       readm = .false.
03679       if (istep==1 .or. (mday==midmonth .and. sec==0)) readm = .true.
03680 
03681 
03682       if (my_task == master_task .and. istep == 1) then
03683          write (nu_diag,*) ' '
03684          write (nu_diag,*) 'SST data interpolated to timestep:'
03685          write (nu_diag,*) trim(ocn_data_dir)//'MONTHLY/sst.1997.nc'
03686          if (restore_sst) write (nu_diag,*) &
03687               'SST restoring timescale (days) =', trestore
03688          if (trim(sst_data_type)=='hadgem_sst_uvocn') then
03689             write (nu_diag,*) ' '
03690             write (nu_diag,*) 'uocn and vocn interpolated to timestep:'
03691             write (nu_diag,*) trim(ocn_data_dir)//'MONTHLY/uocn.1997.nc'
03692             write (nu_diag,*) trim(ocn_data_dir)//'MONTHLY/vocn.1997.nc'
03693          endif
03694       endif                     ! my_task, istep
03695 
03696 
03697       ! -----------------------------------------------------------
03698       ! SST
03699       ! -----------------------------------------------------------
03700       sst_file = trim(ocn_data_dir)//'MONTHLY/sst.1997.nc'      
03701       fieldname='sst'
03702       call read_data_nc (readm, 0, fyear, ixm, month, ixp, &
03703                       maxrec, sst_file, fieldname, sst_data, &
03704                       field_loc_center, field_type_scalar)
03705       
03706       ! Interpolate to current time step
03707       call interpolate_data (sst_data, sstdat)
03708 
03709       ! Restore SSTs if required
03710         if (restore_sst) then
03711          do iblk = 1, nblocks
03712             do j = 1, ny_block
03713             do i = 1, nx_block
03714                sst(i,j,iblk) = sst(i,j,iblk)  &
03715                          + (sstdat(i,j,iblk)-sst(i,j,iblk))*dt/trest
03716             enddo
03717             enddo
03718          enddo
03719          endif      
03720 
03721 
03722       ! -----------------------------------------------------------
03723       ! Ocean currents
03724       ! --------------
03725       ! Values read in are on T grid and oriented geographically, hence 
03726       ! vectors need to be rotated to model grid and then interpolated
03727       ! to U grid.   
03728       ! Also need to be converted from cm s-1 (UM) to m s-1 (CICE)
03729       ! -----------------------------------------------------------
03730 
03731       if (trim(sst_data_type)=='hadgem_sst_uvocn') then
03732 
03733         filename = trim(ocn_data_dir)//'MONTHLY/uocn.1997.nc'   
03734         fieldname='uocn'
03735         call read_data_nc (readm, 0, fyear, ixm, month, ixp, &
03736                       maxrec, filename, fieldname, uocn_data, &
03737                       field_loc_center, field_type_vector)
03738       
03739         ! Interpolate to current time step
03740         call interpolate_data (uocn_data, uocn)
03741 
03742         filename = trim(ocn_data_dir)//'MONTHLY/vocn.1997.nc'   
03743         fieldname='vocn'
03744         call read_data_nc (readm, 0, fyear, ixm, month, ixp, &
03745                       maxrec, filename, fieldname, vocn_data, &
03746                       field_loc_center, field_type_vector)
03747       
03748         ! Interpolate to current time step
03749         call interpolate_data (vocn_data, vocn)
03750 
03751      !----------------------------------------------------------------- 
03752      ! Rotate zonal/meridional vectors to local coordinates, 
03753      ! and change  units
03754      !----------------------------------------------------------------- 
03755 
03756         do iblk = 1, nblocks
03757             do j = 1, ny_block
03758             do i = 1, nx_block
03759 
03760                workx      = uocn(i,j,iblk) 
03761                worky      = vocn(i,j,iblk)
03762                uocn(i,j,iblk) = workx*cos(ANGLET(i,j,iblk)) & 
03763                                   + worky*sin(ANGLET(i,j,iblk))   
03764                vocn(i,j,iblk) = worky*cos(ANGLET(i,j,iblk)) & 
03765                                   - workx*sin(ANGLET(i,j,iblk))
03766 
03767                 uocn(i,j,iblk) = uocn(i,j,iblk) * cm_to_m
03768                 vocn(i,j,iblk) = vocn(i,j,iblk) * cm_to_m
03769 
03770             enddo               ! i
03771             enddo               ! j
03772         enddo           ! nblocks
03773 
03774      !----------------------------------------------------------------- 
03775      ! Interpolate to U grid 
03776      !----------------------------------------------------------------- 
03777 
03778         call t2ugrid_vector(uocn)
03779         call t2ugrid_vector(vocn)
03780 
03781 
03782      endif    !   sst_data_type = hadgem_sst_uvocn
03783 
03784 
03785      end subroutine ocn_data_hadgem
