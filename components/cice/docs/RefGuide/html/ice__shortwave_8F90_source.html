01022            ! SW absorbed in ice interior (nilyr = 1)
01023            fswint(i,j)   = c0
01024            Iswabs(i,j,1) = c0
01025 
01026         enddo                     ! ij
01027 
01028       endif                       ! heat_capacity

01771 ! surface type: either snow over sea ice, bare sea ice, or ponded sea ice.

01773 ! Divides solar spectrum into 3 intervals: 0.2-0.7, 0.7-1.19, and
01774 ! 1.19-5.0 micro-meters. The latter two are added (using an assumed
01775 ! partition of incident shortwave in the 0.7-5.0 micro-meter band between
01776 ! the 0.7-1.19 and 1.19-5.0 micro-meter band) to give the final output 
01777 ! of 0.2-0.7 visible and 0.7-5.0 near-infrared albedos and fluxes.
01778 !
01779 ! Specifies vertical layer optical properties based on input snow depth,
01780 ! density and grain radius, along with ice and pond depths, then computes
01781 ! layer by layer Delta-Eddington reflectivity, transmissivity and combines
01782 ! layers (done by calling routine solution_dEdd). Finally, surface albedos
01783 ! and internal fluxes/flux divergences are evaluated.
01784 !
01785 !  Description of the level and layer index conventions. This is
01786 !  for the standard case of one snow layer and four sea ice layers.
01787 !
01788 !  Please read the following; otherwise, there is 99.9% chance you
01789 !  will be confused about indices at some point in time........ :)
01790 !
01791 !  CICE4.0 snow treatment has one snow layer above the sea ice. This 
01792 !  snow layer has finite heat capacity, so that surface absorption must
01793 !  be distinguished from internal. The Delta-Eddington solar radiation
01794 !  thus adds extra surface scattering layers to both snow and sea ice.
01795 !  Note that in the following, we assume a fixed vertical layer structure
01796 !  for the radiation calculation. In other words, we always have the 
01797 !  structure shown below for one snow and four sea ice layers, but for 
01798 !  ponded ice the pond fills "snow" layer 1 over the sea ice, and for 
01799 !  bare sea ice the top layers over sea ice are treated as transparent air.
01800 !
01801 !  SSL = surface scattering layer for either snow or sea ice
01802 !  DL  = drained layer for sea ice immediately under sea ice SSL
01803 !  INT = interior layers for sea ice below the drained layer.
01804 !
01805 !  Notice that the radiation level starts with 0 at the top. Thus,
01806 !  the total number radiation layers is klev+1, where klev is the
01807 !  sum of nslyr, the number of CCSM snow layers, and nilyr, the
01808 !  number of CCSM sea ice layers, plus the sea ice SSL:
01809 !  klev = 1 + nslyr + nilyr
01810 !
01811 !  For the standard case illustrated below, nslyr=1, nilyr=4,
01812 !  and klev=6, with the number of layer interfaces klevp=klev+1.
01813 !  Layer interfaces are the surfaces on which reflectivities,
01814 !  transmissivities and fluxes are evaluated.
01815 !
01816 !  CCSM3 Sea Ice Model            Delta-Eddington Solar Radiation
01817 !                                     Layers and Interfaces
01818 !                             Layer Index             Interface Index
01819 !    ---------------------            ---------------------  0
01820 !                                  0  \\\   snow SSL    \\\
01821 !       snow layer 1                  ---------------------  1
01822 !                                  1    rest of snow layer
01823 !    +++++++++++++++++++++            +++++++++++++++++++++  2
01824 !                                  2  \\\ sea ice SSL   \\\
01825 !      sea ice layer 1                ---------------------  3
01826 !                                  3      sea ice  DL
01827 !    ---------------------            ---------------------  4
01828 !
01829 !      sea ice layer 2             4      sea ice INT
01830 !

01833 !      sea ice layer 3             5      sea ice INT
01834 !
01835 !    ---------------------            ---------------------  6
01836 !
01837 !      sea ice layer 4             6      sea ice INT
01838 !

01841 ! When snow lies over sea ice, the radiation absorbed in the
01842 ! snow SSL is used for surface heating, and that in the rest
01843 ! of the snow layer for its internal heating. For sea ice in
01844 ! this case, all of the radiant heat absorbed in both the
01845 ! sea ice SSL and the DL are used for sea ice layer 1 heating.
01846 !
01847 ! When pond lies over sea ice, and for bare sea ice, all of the
01848 ! radiant heat absorbed within and above the sea ice SSL is used
01849 ! for surface heating, and that absorbed in the sea ice DL is
01850 ! used for sea ice layer 1 heating.

01852 ! Basically, vertical profiles of the layer extinction optical depth (tau), 
01853 ! single scattering albedo (w0) and asymmetry parameter (g) are required over
01854 ! the klev+1 layers, where klev+1 = 2 + nslyr + nilyr. All of the surface type
01855 ! information and snow/ice iop properties are evaulated in this routine, so
01856 ! the tau,w0,g profiles can be passed to solution_dEdd for multiple scattering

02856                   km = k  - 1
02857                   kp = km + 2
02858                 endif
02859               endif
02860               ki = ki + 1
02861               Iabs(ij,ki) = Iabs(ij,ki) + &
02862               ( ((fdirdn(km,ij)-fdirup(km,ij))*swidr(i,j) + &
02863                  (fdifdn(km,ij)-fdifup(km,ij))*swidf(i,j)) - &
02864                 ((fdirdn(kp,ij)-fdirup(kp,ij))*swidr(i,j) + &
02865                  (fdifdn(kp,ij)-fdifup(kp,ij))*swidf(i,j)) ) &
02866                  *wghtns(ij,ns)
02867             enddo       ! k
02868           enddo        ! ij
02869 
02870         endif        ! ns = 1, ns > 1
02871 
02872       enddo         ! end spectral loop  ns
02873 
02874       ! accumulate fluxes over bare sea ice
02875 !DIR$ CONCURRENT !Cray
02876 !cdir nodep      !NEC
02877 !ocl novrec      !Fujitsu
02878       do ij = 1, icells_DE
02879          i = indxi_DE(ij)
02880          j = indxj_DE(ij)
02881          alvdr(i,j)   = avdr(ij)
02882          alvdf(i,j)   = avdf(ij)
02883          alidr(i,j)   = aidr(ij)
02884          alidf(i,j)   = aidf(ij)
02885          fswsfc(i,j)  = fswsfc(i,j)  + fsfc(ij) *fi(i,j)
02886          fswint(i,j)  = fswint(i,j)  + fint(ij) *fi(i,j)
02887          fswthru(i,j) = fswthru(i,j) + fthru(ij)*fi(i,j)
02888       enddo                     ! ij
02889 
02890       do k = 1, nslyr
02891 !DIR$ CONCURRENT !Cray
02892 !cdir nodep      !NEC
02893 !ocl novrec      !Fujitsu
02894          do ij = 1, icells_DE
02895             i = indxi_DE(ij)
02896             j = indxj_DE(ij)
02897             Sswabs(i,j,k) = Sswabs(i,j,k) + Sabs(ij,k)*fi(i,j)
02898          enddo                  ! ij
02899       enddo                     ! k
02900 
02901       do k = 1, nilyr
02902 !DIR$ CONCURRENT !Cray
02903 !cdir nodep      !NEC
02904 !ocl novrec      !Fujitsu
02905          do ij = 1, icells_DE
02906             i = indxi_DE(ij)
02907             j = indxj_DE(ij)
02908             Iswabs(i,j,k) = Iswabs(i,j,k) + Iabs(ij,k)*fi(i,j)
02909          enddo                  ! ij
02910       enddo                     ! k
02911 
02912       !----------------------------------------------------------------
02913       ! if ice has zero heat capacity, no SW can be absorbed 
02914       ! in the ice/snow interior, so add to surface absorption.
02915       ! Note: nilyr = nslyr = 1 for this case
02916       !----------------------------------------------------------------
02917 
02918       if (.not. heat_capacity) then
02919 
02920 !DIR$ CONCURRENT !Cray
02921 !cdir nodep      !NEC
02922 !ocl novrec      !Fujitsu
02923         do ij = 1, icells_DE
02924            i = indxi_DE(ij)
02925            j = indxj_DE(ij)
02926 
02927            ! SW absorbed at snow/ice surface
02928            fswsfc(i,j) = fswsfc(i,j) + Iswabs(i,j,1) + Sswabs(i,j,1)
02929 
02930            ! SW absorbed in ice interior
02931            fswint(i,j)   = c0
02932            Iswabs(i,j,1) = c0
02933            Sswabs(i,j,1) = c0
02934 
02935         enddo                     ! ij
02936 
02937       endif                       ! heat_capacity
02938 
02939       end subroutine compute_dEdd

03000          albodf      ! ocean albedo to diffuse rad

03004       real (kind=dbl_kind), dimension (0:klevp,icells_DE), 
03005          intent(out) :: 
03006          trndir  ,  ! solar beam down transmission from top
03007          trntdr  ,  ! total transmission to direct beam for layers above
03008          trndif  ,  ! diffuse transmission to diffuse beam for layers above
03009          rupdir  ,  ! reflectivity to direct radiation for layers below
03010          rupdif  ,  ! reflectivity to diffuse radiation for layers below
03011          rdndif      ! reflectivity to diffuse radiation for layers above

03014 !-----------------------------------------------------------------------
03015 !
03016 ! Delta-Eddington solution for snow/air/pond over sea ice
03017 !
03018 ! Generic solution for a snow/air/pond input column of klev+1 layers,
03019 ! with srftyp determining at what interface fresnel refraction occurs.
03020 !
03021 ! Computes layer reflectivities and transmissivities, from the top down
03022 ! to the lowest interface using the Delta-Eddington solutions for each
03023 ! layer; combines layers from top down to lowest interface, and from the
03024 ! lowest interface (underlying ocean) up to the top of the column.
03025 !
03026 ! Note that layer diffuse reflectivity and transmissivity are computed
03027 ! by integrating the direct over several gaussian angles. This is
03028 ! because the diffuse reflectivity expression sometimes is negative,
03029 ! but the direct reflectivity is always well-behaved. We assume isotropic
03030 ! radiation in the upward and downward hemispheres for this integration.
03031 !
03032 ! Assumes monochromatic (spectrally uniform) properties across a band
03033 ! for the input optical parameters.
03034 !
03035 ! If total transmission of the direct beam to the interface above a particular 
03036 ! layer is less than trmin, then no further Delta-Eddington solutions are
03037 ! evaluated for layers below.
03038 !
03039 ! The following describes how refraction is handled in the calculation.
03040 !
03041 ! First, we assume that radiation is refracted when entering either
03042 ! sea ice at the base of the surface scattering layer, or water (i.e. melt
03043 ! pond); we assume that radiation does not refract when entering snow, nor 
03044 ! upon entering sea ice from a melt pond, nor upon entering the underlying 
03045 ! ocean from sea ice.
03046 !
03047 ! To handle refraction, we define a "fresnel" layer, which physically
03048 ! is of neglible thickness and is non-absorbing, which can be combined to 
03049 ! any sea ice layer or top of melt pond. The fresnel layer accounts for 
03050 ! refraction of direct beam and associated reflection and transmission for
03051 ! solar radiation. A fresnel layer is combined with the top of a melt pond 
03052 ! or to the surface scattering layer of sea ice if no melt pond lies over it. 
03053 !
03054 ! Some caution must be exercised for the fresnel layer, because any layer
03055 ! to which it is combined is no longer a homogeneous layer, as are all other
03056 ! individual layers. For all other layers for example, the direct and diffuse
03057 ! reflectivities/transmissivities (R/T) are the same for radiation above or
03058 ! below the layer. This is the meaning of homogeneous! But for the fresnel
03059 ! layer this is not so. Thus, the R/T for this layer must be distinguished
03060 ! for radiation above from that from radiation below. For generality, we
03061 ! treat all layers to be combined as inhomogeneous.
03062 !
03063 !-----------------------------------------------------------------------
03064 
03065 ! Local
03066 
03067       integer (kind=int_kind) ::  
03068          kfrsnl      ! radiation interface index for fresnel layer
03069  
03070       ! following variables are defined for each layer; 0 refers to the top
03071       ! layer. In general we must distinguish directions above and below in 
03072       ! the diffuse reflectivity and transmissivity, as layers are not assumed
03073       ! to be homogeneous (apart from the single layer Delta-Edd solutions); 
03074       ! the direct is always from above.
03075       real (kind=dbl_kind), dimension (0:klev,icells_DE) :: 
03076          rdir    ,  ! layer reflectivity to direct radiation
03077          rdif_a  ,  ! layer reflectivity to diffuse radiation from above
03078          rdif_b  ,  ! layer reflectivity to diffuse radiation from below
03079          tdir    ,  ! layer transmission to direct radiation (solar beam + diffuse)
03080          tdif_a  ,  ! layer transmission to diffuse radiation from above
03081          tdif_b  ,  ! layer transmission to diffuse radiation from below
03082          trnlay      ! solar beam transm for layer (direct beam only)
03083  
03084       integer (kind=int_kind) ::  
03085          i       ,  ! longitude index
03086          j       ,  ! latitude index
03087          ij      ,  ! longitude/latitude index
03088          k           ! level index
03089  
03090       real (kind=dbl_kind), parameter :: 
03091          trmin = 0.001_dbl_kind   ! minimum total transmission allowed
03092       ! total transmission is that due to the direct beam; i.e. it includes
03093       ! both the directly transmitted solar beam and the diffuse downwards
03094       ! transmitted radiation resulting from scattering out of the direct beam 
03095       real (kind=dbl_kind) :: 
03096          tautot   ,  ! layer optical depth
03097          wtot     ,  ! layer single scattering albedo
03098          gtot     ,  ! layer asymmetry parameter
03099          ftot     ,  ! layer forward scattering fraction
03100          ts       ,  ! layer scaled extinction optical depth
03101          ws       ,  ! layer scaled single scattering albedo
03102          gs       ,  ! layer scaled asymmetry parameter
03103          rintfc   ,  ! reflection (multiple) at an interface
03104          refkp1   ,  ! interface multiple scattering for k+1
03105          refkm1   ,  ! interface multiple scattering for k-1
03106          tdrrdir  ,  ! direct tran times layer direct ref 
03107          tdndif       ! total down diffuse = tot tran - direct tran
03108  
03109       ! perpendicular and parallel relative to plane of incidence and scattering
03110       real (kind=dbl_kind) :: 
03111          R1       ,  ! perpendicular polarization reflection amplitude
03112          R2       ,  ! parallel polarization reflection amplitude
03113          T1       ,  ! perpendicular polarization transmission amplitude
03114          T2       ,  ! parallel polarization transmission amplitude
03115          Rf_dir_a ,  ! fresnel reflection to direct radiation
03116          Tf_dir_a ,  ! fresnel transmission to direct radiation
03117          Rf_dif_a ,  ! fresnel reflection to diff radiation from above
03118          Rf_dif_b ,  ! fresnel reflection to diff radiation from below
03119          Tf_dif_a ,  ! fresnel transmission to diff radiation from above
03120          Tf_dif_b     ! fresnel transmission to diff radiation from below
03121  
03122       ! refractive index for sea ice, water; pre-computed, band-independent,
03123       ! diffuse fresnel reflectivities
03124       real (kind=dbl_kind), parameter ::  
03125          refindx = 1.310_dbl_kind  ,  ! refractive index of sea ice (used for water also)
03126          cp063   = 0.063_dbl_kind  ,  ! diffuse fresnel reflectivity from above
03127          cp455   = 0.455_dbl_kind      ! diffuse fresnel reflectivity from below
03128  
03129       real (kind=dbl_kind) :: 
03130          mu0      ,  ! cosine solar zenith angle incident
03131          mu0n         ! cosine solar zenith angle in medium
03132  
03133       real (kind=dbl_kind) :: 
03134          alpha    ,  ! term in direct reflectivity and transmissivity
03135          gamma    ,  ! term in direct reflectivity and transmissivity
03136          el       ,  ! term in alpha,gamma,n,u
03137          taus     ,  ! scaled extinction optical depth
03138          omgs     ,  ! scaled single particle scattering albedo
03139          asys     ,  ! scaled asymmetry parameter
03140          u        ,  ! term in diffuse reflectivity and transmissivity
03141          n        ,  ! term in diffuse reflectivity and transmissivity
03142          lm       ,  ! temporary for el
03143          mu       ,  ! cosine solar zenith for either snow or water
03144          ne           ! temporary for n
03145  
03146       real (kind=dbl_kind) :: 
03147          w        ,  ! dummy argument for statement function
03148          uu       ,  ! dummy argument for statement function
03149          gg       ,  ! dummy argument for statement function
03150          e        ,  ! dummy argument for statement function
03151          f        ,  ! dummy argument for statement function
03152          t        ,  ! dummy argument for statement function
03153          et           ! dummy argument for statement function
03154  
03155       real (kind=dbl_kind) :: 
03156          alp      ,  ! temporary for alpha
03157          gam      ,  ! temporary for gamma
03158          ue       ,  ! temporary for u
03159          arg      ,  ! exponential argument
03160          extins   ,  ! extinction
03161          amg      ,  ! alp - gam
03162          apg          ! alp + gam
03163  
03164       integer (kind=int_kind), parameter :: 
03165          ngmax = 8    ! number of gaussian angles in hemisphere
03166  
03167       real (kind=dbl_kind), dimension (ngmax) :: 
03168          gauspt   ,  ! gaussian angles (radians)
03169          gauswt       ! gaussian weights
03170 
03171       data gauspt/ &
03172                  .9894009_dbl_kind,  .9445750_dbl_kind, &
03173                  .8656312_dbl_kind,  .7554044_dbl_kind, &
03174                  .6178762_dbl_kind,  .4580168_dbl_kind, &
03175                  .2816036_dbl_kind,  .0950125_dbl_kind  /
03176       data gauswt/ &     
03177                  .0271525_dbl_kind,  .0622535_dbl_kind, &
03178                  .0951585_dbl_kind,  .1246290_dbl_kind, &
03179                  .1495960_dbl_kind,  .1691565_dbl_kind, &
03180                  .1826034_dbl_kind,  .1894506_dbl_kind  /
03181   
03182       integer (kind=int_kind) :: 
03183          ng           ! gaussian integration index
03184  
03185       real (kind=dbl_kind) :: 
03186          gwt      ,  ! gaussian weight
03187          swt      ,  ! sum of weights
03188          trn      ,  ! layer transmission
03189          rdr      ,  ! rdir for gaussian integration
03190          tdr      ,  ! tdir for gaussian integration
03191          smr      ,  ! accumulator for rdif gaussian integration
03192          smt          ! accumulator for tdif gaussian integration
03193  
03194       ! Delta-Eddington solution expressions
03195       alpha(w,uu,gg,e) = p75*w*uu*((c1 + gg*(c1-w))/(c1 - e*e*uu*uu))
03196       gamma(w,uu,gg,e) = p5*w*((c1 + c3*gg*(c1-w)*uu*uu) &
03197                         / (c1-e*e*uu*uu))
03198       n(uu,et)         = ((uu+c1)*(uu+c1)/et ) - ((uu-c1)*(uu-c1)*et)
03199       u(w,gg,e)        = c1p5*(c1 - w*gg)/e
03200       el(w,gg)         = sqrt(c3*(c1-w)*(c1 - w*gg))
03201       taus(w,f,t)      = (c1 - w*f)*t
03202       omgs(w,f)        = (c1 - f)*w/(c1 - w*f)
03203       asys(gg,f)       = (gg - f)/(c1 - f)
03204  
03205 !-----------------------------------------------------------------------
03206 
03207       ! initialize all output to 0
03208       do ij = 1, icells_DE
03209       do k = 0, klevp 
03210          trndir(k,ij) = c0
03211          trntdr(k,ij) = c0
03212          trndif(k,ij) = c0
03213          rupdir(k,ij) = c0
03214          rupdif(k,ij) = c0
03215          rdndif(k,ij) = c0
03216       enddo
03217  
03218       ! initialize all layer apparent optical properties to 0
03219       do k = 0, klev
03220          rdir(k,ij)   = c0
03221          rdif_a(k,ij) = c0
03222          rdif_b(k,ij) = c0
03223          tdir(k,ij)   = c0
03224          tdif_a(k,ij) = c0
03225          tdif_b(k,ij) = c0
03226          trnlay(k,ij) = c0
03227       enddo
03228 
03229       ! initialize top interface of top layer 
03230         trndir(0,ij) =   c1
03231         trntdr(0,ij) =   c1
03232         trndif(0,ij) =   c1
03233         rdndif(0,ij) =   c0
03234       enddo       ! ij
03235  
03236       ! proceed down one layer at a time; if the total transmission to
03237       ! the interface just above a given layer is less than trmin, then no
03238       ! Delta-Eddington computation for that layer is done.
03239 
03240       do ij = 1, icells_DE 
03241          i = indxi_DE(ij)
03242          j = indxj_DE(ij)
03243 
03244         ! begin main level loop
03245         do k=0,klev
03246  
03247         ! initialize current layer properties to zero; only if total
03248         ! transmission to the top interface of the current layer exceeds the
03249         ! minimum, will these values be computed below:
03250         if ( k > 0 ) then  
03251             ! Calculate the solar beam transmission, total transmission, and
03252             ! reflectivity for diffuse radiation from below at interface k, 
03253             ! the top of the current layer k:
03254             !
03255             !              layers       interface
03256             !         
03257             !       ---------------------  k-1 
03258             !                k-1
03259             !       ---------------------  k
03260             !                 k
03261             !       ---------------------  
03262 
03263               trndir(k,ij) = trndir(k-1,ij)*trnlay(k-1,ij)
03264               refkm1        = c1/(c1 - rdndif(k-1,ij)*rdif_a(k-1,ij))
03265               tdrrdir       = trndir(k-1,ij)*rdir(k-1,ij)
03266               tdndif        = trntdr(k-1,ij) - trndir(k-1,ij)
03267               trntdr(k,ij) = trndir(k-1,ij)*tdir(k-1,ij) + &
03268                (tdndif + tdrrdir*rdndif(k-1,ij))*refkm1*tdif_a(k-1,ij)
03269               rdndif(k,ij) = rdif_b(k-1,ij) + &
03270                 (tdif_b(k-1,ij)*rdndif(k-1,ij)*refkm1*tdif_a(k-1,ij))
03271               trndif(k,ij) = trndif(k-1,ij)*refkm1*tdif_a(k-1,ij)
03272         endif       ! k > 0  
03273  
03274         ! compute next layer Delta-eddington solution only if total transmission
03275         ! of radiation to the interface just above the layer exceeds trmin.
03276           if (trntdr(k,ij) > trmin ) then
03277 
03278         ! calculation over layers with penetrating radiation
03279  
03280            tautot  = tau(k,ij)
03281            wtot    = w0(k,ij)
03282            gtot    = g(k,ij)
03283            ftot    = gtot*gtot
03284  
03285            ts   = taus(wtot,ftot,tautot)
03286            ws   = omgs(wtot,ftot)
03287            gs   = asys(gtot,ftot)
03288            lm   = el(ws,gs)
03289            ue   = u(ws,gs,lm)
03290 
03291            ! compute level of fresnel refraction
03292            if( srftyp(i,j) < 2 ) then
03293              ! if snow over sea ice or bare sea ice, fresnel level is
03294              ! at base of sea ice SSL (and top of the sea ice DL); the
03295              ! snow SSL counts for one, then the number of snow layers,
03296              ! then the sea ice SSL which also counts for one:
03297              kfrsnl = nslyr + 2 
03298            else
03299              ! if ponded sea ice, fresnel level is the top of the pond 
03300              kfrsnl = 0
03301            endif
03302 
03303            ! mu0 is cosine solar zenith angle above the fresnel level; make 
03304            ! sure mu0 is large enough for stable and meaningful radiation
03305            ! solution: .01 is like sun just touching horizon with its lower edge
03306 
03307            mu0  = max(coszen(i,j),p01)
03308 
03309            ! mu0n is cosine solar zenith angle used to compute the layer
03310            ! Delta-Eddington solution; it is initially computed to be the
03311            ! value below the fresnel level, i.e. the cosine solar zenith 
03312            ! angle below the fresnel level for the refracted solar beam:
03313 
03314            mu0n = sqrt(c1-((c1-mu0*mu0)/(refindx*refindx)))
03315 
03316            ! if level k is above fresnel level and the cell is non-pond, use the
03317            ! non-refracted beam instead
03318 
03319            if( srftyp(i,j) < 2 .and. k < kfrsnl ) mu0n = mu0
03320  
03321            extins = max(exp_min, exp(-lm*ts))
03322            ne = n(ue,extins)
03323  
03324            ! first calculation of rdif, tdif using Delta-Eddington formulas
03325 
03326            rdif_a(k,ij) = (ue+c1)*(ue-c1)*(c1/extins - extins)/ne
03327            tdif_a(k,ij) = c4*ue/ne
03328  
03329            ! evaluate rdir,tdir for direct beam
03330            trnlay(k,ij) = max(exp_min, exp(-ts/mu0n))
03331            alp = alpha(ws,mu0n,gs,lm)
03332            gam = gamma(ws,mu0n,gs,lm)
03333            apg = alp + gam
03334            amg = alp - gam
03335            rdir(k,ij) = amg*(tdif_a(k,ij)*trnlay(k,ij) - c1) + &
03336                        apg*rdif_a(k,ij)
03337            tdir(k,ij) = apg*tdif_a(k,ij) + &
03338                        (amg*rdif_a(k,ij) - (apg-c1))*trnlay(k,ij)
03339  
03340            ! recalculate rdif,tdif using direct angular integration over rdir,tdir,
03341            ! since Delta-Eddington rdif formula is not well-behaved (it is usually
03342            ! biased low and can even be negative); use ngmax angles and gaussian
03343            ! integration for most accuracy:
03344            swt = c0
03345            smr = c0
03346            smt = c0
03347            do ng=1,ngmax
03348              mu  = gauspt(ng)
03349              gwt = gauswt(ng)
03350              swt = swt + mu*gwt
03351              trn = max(exp_min, exp(-ts/mu))
03352              alp = alpha(ws,mu,gs,lm)
03353              gam = gamma(ws,mu,gs,lm)
03354              apg = alp + gam
03355              amg = alp - gam
03356              rdr = amg*(tdif_a(k,ij)*trn-c1) + &
03357                    apg*rdif_a(k,ij)
03358              tdr = apg*tdif_a(k,ij) + &
03359                    (amg*rdif_a(k,ij)-(apg-c1))*trn
03360              smr = smr + mu*rdr*gwt
03361              smt = smt + mu*tdr*gwt
03362            enddo      ! ng
03363            rdif_a(k,ij) = smr/swt
03364            tdif_a(k,ij) = smt/swt
03365  
03366            ! homogeneous layer
03367            rdif_b(k,ij) = rdif_a(k,ij)
03368            tdif_b(k,ij) = tdif_a(k,ij)
03369  
03370            ! add fresnel layer to top of desired layer if either 
03371            ! air or snow overlies ice; we ignore refraction in ice 
03372            ! if a melt pond overlies it:
03373 
03374            if( k == kfrsnl ) then
03375              ! compute fresnel reflection and transmission amplitudes
03376              ! for two polarizations: 1=perpendicular and 2=parallel to
03377              ! the plane containing incident, reflected and refracted rays.
03378              R1 = (mu0 - refindx*mu0n) / &

03422              tdif_b(kfrsnl,ij) = tdif_b(kfrsnl,ij)*rintfc*Tf_dif_b

03425              trnlay(kfrsnl,ij) = Tf_dir_a*trnlay(kfrsnl,ij)

03433       ! reflectivity for diffuse radiation (from below) for all layers
03434       ! above the underlying ocean; note that we ignore refraction between 
03435       ! sea ice and underlying ocean:
03436       !
03437       !       For k = klevp
03438       !
03439       !              layers       interface
03440       !
03441       !       ---------------------  k-1 
03442       !                k-1
03443       !       ---------------------  k
03444       !       \\\\\\\ ocean \\\\\\\

03446         k = klevp
03447         trndir(k,ij) = trndir(k-1,ij)*trnlay(k-1,ij)
03448         refkm1        = c1/(c1 - rdndif(k-1,ij)*rdif_a(k-1,ij))
03449         tdrrdir       = trndir(k-1,ij)*rdir(k-1,ij)
03450         tdndif        = trntdr(k-1,ij) - trndir(k-1,ij)
03451         trntdr(k,ij) = trndir(k-1,ij)*tdir(k-1,ij) + &
03452           (tdndif + tdrrdir*rdndif(k-1,ij))*refkm1*tdif_a(k-1,ij)
03453         rdndif(k,ij) = rdif_b(k-1,ij) + &
03454           (tdif_b(k-1,ij)*rdndif(k-1,ij)*refkm1*tdif_a(k-1,ij))
03455         trndif(k,ij) = trndif(k-1,ij)*refkm1*tdif_a(k-1,ij)
03456  
03457       ! compute reflectivity to direct and diffuse radiation for layers 
03458       ! below by adding succesive layers starting from the underlying 
03459       ! ocean and working upwards:
03460       !
03461       !              layers       interface
03462       !
03463       !       ---------------------  k
03464       !                 k
03465       !       ---------------------  k+1
03466       !                k+1
03467       !       ---------------------
03468 
03469         rupdir(klevp,ij) = albodr(ij)
03470         rupdif(klevp,ij) = albodf(ij)

03472       do k=klev,0,-1
03473         do ij = 1, icells_DE
03474           i = indxi_DE(ij)
03475           j = indxj_DE(ij)
03476           ! interface scattering
03477           refkp1        = c1/( c1 - rdif_b(k,ij)*rupdif(k+1,ij))
03478           ! dir from top layer plus exp tran ref from lower layer, interface
03479           ! scattered and tran thru top layer from below, plus diff tran ref
03480           ! from lower layer with interface scattering tran thru top from below
03481           rupdir(k,ij) = rdir(k,ij) + &
03482                         ( trnlay(k,ij)*rupdir(k+1,ij) + &

03488                           tdif_a(k,ij)*rupdif(k+1,ij)* &
03489                           refkp1*tdif_b(k,ij)
03490         enddo       ! ij
