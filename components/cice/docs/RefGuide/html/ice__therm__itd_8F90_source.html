00567             if (dvice(m,n) < vicen(i,j,nd)*puny) then
00568                daice(m,n) = c0
00569                dvice(m,n) = c0
00570                donor(m,n) = 0
00571             endif
00572 
00573             ! If daice is close to aicen or dvice is close to vicen,
00574             ! shift entire category
00575 
00576             if (daice(m,n) > aicen(i,j,nd)*(c1-puny)) then
00577                daice(m,n) = aicen(i,j,nd)
00578                dvice(m,n) = vicen(i,j,nd)
00579             endif
00580 
00581             if (dvice(m,n) > vicen(i,j,nd)*(c1-puny)) then
00582                daice(m,n) = aicen(i,j,nd)
00583                dvice(m,n) = vicen(i,j,nd)
00584             endif
00585 
00586          enddo                  ! ij
00587       enddo                     ! boundaries, 1 to ncat-1
00588 
00589       deallocate(g0)
00590       deallocate(g1)
00591       deallocate(hL)
00592       deallocate(hR)
00593 
00594       !-----------------------------------------------------------------
00595       ! Shift ice between categories as necessary
00596       !-----------------------------------------------------------------
00597 
00598       call shift_ice (nx_block, ny_block,    &
00599                       indxi,    indxj,       &
00600                       icells,   trcr_depend, &
00601                       aicen,    trcrn,       &
00602                       vicen,    vsnon,       &
00603                       eicen,    esnon,       &
00604                       hicen,    donor,       &
00605                       daice,    dvice,       &
00606                       l_stop,                &
00607                       istop,    jstop)
00608 
00609       if (l_stop) return
00610 
00611       !-----------------------------------------------------------------
00612       ! Make sure hice(i,j,1) >= minimum ice thickness hi_min.
00613       !-----------------------------------------------------------------
00614 
00615 !DIR$ CONCURRENT !Cray
00616 !cdir nodep      !NEC
00617 !ocl novrec      !Fujitsu
00618       do ij = 1, iflag          ! remap_flag = .true.
00619          i = indxii(ij)
00620          j = indxjj(ij)
00621          m = indxij(ij)
00622          if (hi_min > c0 .and. &
00623               aicen(i,j,1) > puny .and. hicen(m,1) < hi_min) then
00624             aicen(i,j,1) = aicen(i,j,1) * hicen(m,1)/hi_min
00625             hicen(m,1) = hi_min
00626          endif
00627       enddo                     ! ij
00628 
00629       !-----------------------------------------------------------------
00630       ! Update fractional ice area in each grid cell.
00631       !-----------------------------------------------------------------
00632       call aggregate_area (nx_block, ny_block, &
00633                            aicen,              &
00634                            aice,     aice0)
00635 
00636       if (l_stop) return
00637 
00638       !-----------------------------------------------------------------
00639       ! Check volume and energy conservation.
00640       !-----------------------------------------------------------------
00641 
00642       if (l_conservation_check) then
00643 
00644          call column_sum (nx_block, ny_block,       &
00645                           icells,   indxi,   indxj, &
00646                           ncat,                     &
00647                           vicen,    vice_final)
00648          fieldid = 'vice, ITD remap'
00649          call column_conservation_check (nx_block,  ny_block,      &
00650                                          icells,   indxi,   indxj, &
00651                                          fieldid,                  &
00652                                          vice_init, vice_final,    &
00653                                          puny,      l_stop,        &
00654                                          istop,     jstop)
00655          if (l_stop) return
00656 
00657          call column_sum (nx_block, ny_block,       &
00658                           icells,   indxi,   indxj, &
00659                           ncat,                     &
00660                           vsnon,    vsno_final)
00661          fieldid = 'vsno, ITD remap'
00662          call column_conservation_check (nx_block,  ny_block,      &
00663                                          icells,   indxi,   indxj, &
00664                                          fieldid,                  &
00665                                          vsno_init, vsno_final,    &
00666                                          puny,      l_stop,        &
00667                                          istop,     jstop)
00668          if (l_stop) return
00669 
00670          call column_sum (nx_block, ny_block,       &
00671                           icells,   indxi,   indxj, &
00672                           ntilyr,                   &
00673                           eicen,    eice_final)
00674          fieldid = 'eice, ITD remap'
00675          call column_conservation_check (nx_block,   ny_block,     &
00676                                          icells,   indxi,   indxj, &
00677                                          fieldid,                  &
00678                                          eice_init,  eice_final,   &
00679                                          puny*Lfresh*rhoi,         &
00680                                          l_stop,                   &
00681                                          istop,     jstop)
00682          if (l_stop) return
00683 
00684          call column_sum (nx_block, ny_block,       &
00685                           icells,   indxi,   indxj, &
00686                           ntslyr,                   &
00687                           esnon,    esno_final)
00688          fieldid = 'esno, ITD remap'
00689          call column_conservation_check (nx_block,   ny_block,     &
00690                                          icells,   indxi,   indxj, &
00691                                          fieldid,                  &
00692                                          esno_init,  esno_final,   &
00693                                          puny*Lfresh*rhos,         &
00694                                          l_stop,                   &
00695                                          istop,     jstop)
00696          if (l_stop) return
00697 
00698       endif                     ! conservation check

00702 !=======================================================================
00703 !BOP
00704 !
00705 ! !IROUTINE: fit_line - fit g(h) with a line using area, volume constraints
00706 !
00707 ! !INTERFACE:
00708 !
00709 
00710       subroutine fit_line (nx_block, ny_block,        &
00711                            iflag,   icells,           &
00712                            indxii,   indxjj,  indxij, &
00713                            aicen,    hice,            &
00714                            hbL,      hbR,             &
00715                            g0,       g1,              &
00716                            hL,       hR)
00717 !
00718 ! !DESCRIPTION:
00719 !
00720 ! Fit g(h) with a line, satisfying area and volume constraints.
00721 ! To reduce roundoff errors caused by large values of g0 and g1,
00722 ! we actually compute g(eta), where eta = h - hL, and hL is the
00723 ! left boundary.
00724 !
00725 ! !REVISION HISTORY:
00726 !
00727 ! authors: William H. Lipscomb, LANL
00728 !          Elizabeth C. Hunke, LANL
00729 !
00730 ! !USES:
00731 !
00732 ! !INPUT/OUTPUT PARAMETERS:
00733 !
00734       integer (kind=int_kind), intent(in) :: 
00735          nx_block, ny_block,  ! block dimensions
00736          icells            ,  ! number of grid cells with ice
00737          iflag                 ! number of grid cells with remap_flag = .true.
00738 
00739        integer (kind=int_kind), dimension (icells), 
00740          intent(in) :: 
00741          indxii, indxjj,  ! compressed i/j indices (from iflag)
00742          indxij            ! compressed i/j indices (from icells)
00743 
00744       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: 
00745          aicen           ! concentration of ice
00746 
00747       real (kind=dbl_kind), dimension (icells), intent(in) :: 
00748          hbL, hbR    ,  ! left and right category boundaries
00749          hice            ! ice thickness
00750 
00751       real (kind=dbl_kind), dimension (iflag), intent(out):: 
00752          g0, g1      ,  ! coefficients in linear equation for g(eta)
00753          hL          ,  ! min value of range over which g(h) > 0
00754          hR              ! max value of range over which g(h) > 0
00755 !
00756 !EOP
00757 !
00758       integer (kind=int_kind) :: 
00759          i,j         ,  ! horizontal indices
00760          ij, m           ! combined horizontal indices
00761 
00762       real  (kind=dbl_kind) :: 
00763          h13         ,  ! hbL + 1/3 * (hbR - hbL)
00764          h23         ,  ! hbL + 2/3 * (hbR - hbL)
00765          dhr         ,  ! 1 / (hR - hL)
00766          wk1, wk2        ! temporary variables
00767 
00768       !-----------------------------------------------------------------
00769       ! Compute g0, g1, hL, and hR for each category to be remapped.
00770       !-----------------------------------------------------------------
00771 
00772        do ij = 1, iflag
00773          i = indxii(ij)
00774          j = indxjj(ij)
00775          m = indxij(ij)
00776 
00777          if (aicen(i,j) > puny .and. hbR(m) - hbL(m) > puny) then
00778 
00779          ! Initialize hL and hR
00780 
00781             hL(ij) = hbL(m)
00782             hR(ij) = hbR(m)
00783 
00784          ! Change hL or hR if hicen(n) falls outside central third of range
00785 
00786             h13 = p333 * (c2*hL(ij) + hR(ij))
00787             h23 = p333 * (hL(ij) + c2*hR(ij))
00788             if (hice(m) < h13) then
00789                hR(ij) = c3*hice(m) - c2*hL(ij)
00790             elseif (hice(m) > h23) then
00791                hL(ij) = c3*hice(m) - c2*hR(ij)
00792             endif
00793 
00794          ! Compute coefficients of g(eta) = g0 + g1*eta
00795 
00796             dhr = c1 / (hR(ij) - hL(ij))
00797             wk1 = c6 * aicen(i,j) * dhr
00798             wk2 = (hice(m) - hL(ij)) * dhr
00799             g0(ij) = wk1 * (p666 - wk2)
00800             g1(ij) = c2*dhr * wk1 * (wk2 - p5)
00801 
00802          else
00803 
00804             g0(ij) = c0
00805             g1(ij) = c0
00806             hL(ij) = c0
00807             hR(ij) = c0
00808 
00809          endif                  ! aicen > puny
00810 
00811       enddo                     ! ij
00812 
00813       end subroutine fit_line

00900          fsalt         ! salt flux to ocean (kg/m^2/s)
00901 
00902       real (kind=dbl_kind), dimension (nx_block,ny_block), 
00903          intent(inout), optional :: 
00904          frz_onset ! day of year that freezing begins (congel or frazil)
00905 
00906       real (kind=dbl_kind), intent(in), optional :: 
00907          yday      ! day of year
00908 
00909       logical (kind=log_kind), intent(out) :: 
00910          l_stop    ! if true, abort on return
00911 
00912       integer (kind=int_kind), intent(out) :: 
00913          istop, jstop    ! indices of grid cell where model aborts
00914 !
00915 !EOP
00916 !
00917       integer (kind=int_kind) :: 
00918          i, j         ,  ! horizontal indices
00919          n            ,  ! ice category index
00920          k            ,  ! ice layer index

00922 
00923       real (kind=dbl_kind), dimension (icells) :: 
00924          ai0new       ,  ! area of new ice added to cat 1

00926          hsurp        ,  ! thickness of new ice added to each cat
00927          vlyr             ! ice layer volume
00928 
00929       real (kind=dbl_kind), dimension (icells) :: 
00930          vice_init, vice_final  ! ice volume summed over categories
00931 
00932       real (kind=dbl_kind) :: 
00933          fnew         ,  ! heat flx to open water for new ice (W/m^2)
00934          hi0new       ,  ! thickness of new ice
00935          hi0max       ,  ! max allowed thickness of new ice
00936          qi0(nilyr)   ,  ! frazil ice enthalpy
00937          qi0av        ,  ! mean value of qi0 for new ice (J kg-1)
00938          vsurp        ,  ! volume of new ice added to each cat
00939          area1        ,  ! starting fractional area of existing ice
00940          vice1        ,  ! starting volume of existing ice
00941          rnilyr       ,  ! real(nilyr)
00942          dfresh       ,  ! change in fresh
00943          dfsalt       ,  ! change in fsalt
00944          vtmp
00945 
00946       integer (kind=int_kind) :: 
00947          jcells, kcells     ,  ! grid cell counters
00948          ij, m                  ! combined i/j horizontal indices
00949 
00950       integer (kind=int_kind), dimension (icells) :: 
00951          indxij2,  indxij3  ,  ! compressed i/j indices
00952          indxi2, indxj2     , 
00953          indxi3, indxj3
00954

00957 
00958       l_stop = .false.
00959       istop = 0
00960       jstop = 0
00961 
00962       jcells = 0
00963       kcells = 0
00964 
00965       if (ncat > 1) then
00966          hi0max = hin_max(1)*0.9_dbl_kind  ! not too close to boundary
00967       else
00968          hi0max = bignum                   ! big number
00969       endif
00970 
00971       ! initial ice volume in each grid cell
00972       call column_sum (nx_block, ny_block,       &
00973                        icells,   indxi,   indxj, &
00974                        ncat,                     &
00975                        vicen,    vice_init)
00976 
00977       !-----------------------------------------------------------------
00978       ! Compute average enthalpy of new ice.
00979       !
00980       ! POP assumes new ice is fresh.  Otherwise, it would be better
00981       ! to do something like this:
00982       !  qi0(i,j,k) = -rhoi * (cp_ice*(Tmlt(k)-Tf(i,j))
00983       !             + Lfresh*(1.-Tmlt(k)/Tf(i,j)) - cp_ocn*Tmlt(k))
00984       !-----------------------------------------------------------------
00985 
00986       rnilyr = real(nilyr,kind=dbl_kind)
00987       qi0av = c0
00988       do k = 1, nilyr
00989          qi0(k) = -rhoi*Lfresh  ! note sign convention, qi < 0
00990          qi0av  = qi0av + qi0(k)
00991       enddo
00992       qi0av = qi0av/rnilyr
00993

00995       ! Compute the volume, area, and thickness of new ice.
00996       !-----------------------------------------------------------------

00999 !cdir nodep      !NEC
01000 !ocl novrec      !Fujitsu

01003          j = indxj(ij)
01004

01010

01012          frazil(i,j) = vi0new(ij)
01013

01016                  frz_onset(i,j) = yday
01017          endif

01020       ! Update fresh water and salt fluxes.
01021       !
01022       ! NOTE: POP assumes fresh water and salt flux due to frzmlt > 0
01023       !       is NOT included in fluxes fresh and fsalt.
01024       !-----------------------------------------------------------------
01025 
01026          if (update_ocn_f) then
01027             dfresh = -rhoi*vi0new(ij)/dt
01028             dfsalt = ice_ref_salinity*p001*dfresh
01029             fresh(i,j)      = fresh(i,j)      + dfresh
01030             fsalt(i,j)      = fsalt(i,j)      + dfsalt
01031          endif
01032 
01033       !-----------------------------------------------------------------
01034       ! Decide how to distribute the new ice.
01035       !-----------------------------------------------------------------
01036

01038          ai0new(ij) = c0
01039 
01040          if (vi0new(ij) > c0) then
01041 
01042             ! new ice area and thickness
01043             ! hin_max(0) < new ice thickness < hin_max(1)
01044             if (aice0(i,j) > puny) then

01047                   ! distribute excess volume over all categories (below)
01048                   hi0new = hi0max
01049                   ai0new(ij) = aice0(i,j)
01050                   vsurp       = vi0new(ij) - ai0new(ij)*hi0new
01051                   hsurp(ij)  = vsurp / aice(i,j)
01052                   vi0new(ij) = ai0new(ij)*hi0new
01053                else
01054                   ! put ice in a single category, with hsurp = 0
01055                   ai0new(ij) = vi0new(ij)/hi0new
01056                endif
01057             else                ! aice0 < puny
01058                hsurp(ij) = vi0new(ij)/aice(i,j) ! new thickness in each cat
01059                vi0new(ij) = c0
01060             endif               ! aice0 > puny
01061          endif                  ! vi0new > puny
01062 
01063       !-----------------------------------------------------------------
01064       ! Identify grid cells receiving new ice.
01065       !-----------------------------------------------------------------
01066 
01067          i = indxi(ij)
01068          j = indxj(ij)
01069 
01070          if (vi0new(ij) > c0) then  ! add ice to category 1
01071             jcells = jcells + 1
01072             indxi2(jcells) = i
01073             indxj2(jcells) = j
01074             indxij2(jcells) = ij
01075          endif
01076 
01077          if (hsurp(ij) > c0) then   ! add ice to all categories
01078             kcells = kcells + 1
01079             indxi3(kcells) = i
01080             indxj3(kcells) = j
01081             indxij3(kcells) = ij
01082          endif
01083 
01084       enddo                     ! ij
01085 
01086       !-----------------------------------------------------------------
01087       ! Distribute excess ice volume among ice categories by increasing
01088       ! ice thickness, leaving ice area unchanged.
01089       !
01090       ! NOTE: If new ice contains globally conserved tracers
01091       !       (e.g., isotopes from seawater), code must be added here.
01092       !-----------------------------------------------------------------
01093 
01094       do n = 1, ncat
01095 
01096 !DIR$ CONCURRENT !Cray
01097 !cdir nodep      !NEC
01098 !ocl novrec      !Fujitsu
01099          do ij = 1, kcells
01100             i = indxi3(ij)
01101             j = indxj3(ij)
01102             m = indxij3(ij)
01103 
01104             vsurp = hsurp(m) * aicen(i,j,n)
01105 
01106             ! update ice age due to freezing (new ice age = dt)
01107             vtmp = vicen(i,j,n) + vsurp
01108             if (vtmp > puny) then
01109 
01110             if (tr_iage) trcrn(i,j,nt_iage,n)  &
01111                    = (trcrn(i,j,nt_iage,n)*vicen(i,j,n) + dt*vsurp) &
01112                    / vtmp
01113 
01114             if (tr_aero) then
01115              do it=1,n_aero
01116                trcrn(i,j,nt_aero+2+4*(it-1),n)  &
01117                    = trcrn(i,j,nt_aero+2+4*(it-1),n)*vicen(i,j,n) &
01118                    / vtmp
01119                trcrn(i,j,nt_aero+3+4*(it-1),n)  &
01120                    = trcrn(i,j,nt_aero+3+4*(it-1),n)*vicen(i,j,n) &
01121                    / vtmp
01122              enddo
01123             endif
01124 
01125             endif
01126 
01127             ! update category volumes
01128             vicen(i,j,n) = vicen(i,j,n) + vsurp
01129             vlyr(m) = vsurp/rnilyr
01130 
01131          enddo                  ! ij
01132 
01133          do k = 1, nilyr
01134 !DIR$ CONCURRENT !Cray
01135 !cdir nodep      !NEC
01136 !ocl novrec      !Fujitsu
01137             do ij = 1, kcells
01138                i = indxi3(ij)
01139                j = indxj3(ij)
01140                m = indxij3(ij)
01141 
01142                eicen(i,j,ilyr1(n)+k-1) = &
01143                     eicen(i,j,ilyr1(n)+k-1) + qi0(k)*vlyr(m)
01144             enddo               ! ij
01145          enddo                  ! k
01146 
01147       enddo                     ! n
01148 
01149       !-----------------------------------------------------------------
01150       ! Combine new ice grown in open water with category 1 ice.
01151       ! Assume that vsnon and esnon are unchanged.
01152       !-----------------------------------------------------------------
01153 
01154 !DIR$ CONCURRENT !Cray
01155 !cdir nodep      !NEC
01156 !ocl novrec      !Fujitsu
01157       do ij = 1, jcells
