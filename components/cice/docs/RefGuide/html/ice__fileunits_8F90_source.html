00159 
00160 ! !DESCRIPTION:
00161 !  This routine returns the next available I/O unit and marks it as
00162 !  in use to prevent any later use.
00163 !  Note that {\em all} processors must call this routine even if only
00164 !  the master task is doing the I/O.  This is necessary insure that
00165 !  the units remain synchronized for other parallel I/O functions.
00166 !
00167 ! !REVISION HISTORY:
00168 !  same as module
00169 
00170 ! !OUTPUT PARAMETERS:
00171 
00172    integer (kind=int_kind), intent(out) :: 
00173       iunit                     ! next free I/O unit
00174 !EOP
00175 !BOC
00176 
00177    integer (kind=int_kind) :: n  ! dummy loop index
00178 
00179    logical (kind=log_kind) :: alreadyInUse
00180 
00181 #ifdef CCSMCOUPLED
00182    iunit = shr_file_getUnit()
00183 #else
00184    srch_units: do n=ice_IOUnitsMinUnits, ice_IOUnitsMaxUnits
00185       if (.not. ice_IOUnitsInUse(n)) then   ! I found one, I found one
00186 
00187          !*** make sure not in use by library or calling routines
00188          INQUIRE (unit=n,OPENED=alreadyInUse)
00189 
00190          if (.not. alreadyInUse) then
00191             iunit = n        ! return the free unit number
00192             ice_IOUnitsInUse(n) = .true.  ! mark iunit as being in use
00193             exit srch_units
00194          else
00195             !*** if inquire shows this unit in use, mark it as
00196             !***    in use to prevent further queries
00197             ice_IOUnitsInUse(iunit) = .true.
00198          endif
00199       endif
00200    end do srch_units
00201 
00202    if (iunit > ice_IOUnitsMaxUnits) stop 'ice_IOUnitsGet: No free units'
00203 #endif
00204 
00205 !EOC
00206  end subroutine get_fileunit
