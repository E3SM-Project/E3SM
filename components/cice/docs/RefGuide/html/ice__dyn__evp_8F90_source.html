01377                         icellu,               &
01378                         indxui,     indxuj,   &
01379                         aiu,        str,      &
01380                         uocn,       vocn,     &
01381                         waterx,     watery,   &
01382                         forcex,     forcey,   &
01383                         umassdtei,  fm,       &
01384                         uarear,               &
01385                         strocnx,    strocny,  &
01386                         strintx,    strinty,  &
01387                         uvel,       vvel)
01388 !
01389 ! !DESCRIPTION:
01390 !
01391 ! Calculation of the surface stresses
01392 ! Integration of the momentum equation to find velocity (u,v)
01393 !
01394 ! !REVISION HISTORY:
01395 !
01396 ! author: Elizabeth C. Hunke, LANL
01397 !
01398 ! !USES:
01399 !
01400 ! !INPUT/OUTPUT PARAMETERS:
01401 !
01402       integer (kind=int_kind), intent(in) :: 
01403          nx_block, ny_block,  ! block dimensions
01404          icellu                ! total count when iceumask is true
01405 
01406       integer (kind=int_kind), dimension (nx_block*ny_block), 
01407          intent(in) :: 
01408          indxui  ,  ! compressed index in i-direction
01409          indxuj      ! compressed index in j-direction

01414          watery  ,  ! for ocean stress calculation, y (m/s)

01420          fm      ,  ! Coriolis param. * mass in U-cell (kg/s)

01422

01425          str
01426

01430          vvel        ! y-component of velocity (m/s)
01431

01440 !

01457

01459          vold = vvel(i,j)

01462          vrel = aiu(i,j)*dragw*sqrt((uocn(i,j) - uold)**2 + &
01463                                     (vocn(i,j) - vold)**2)  ! m/s
01464          ! ice/ocean stress
01465          taux = vrel*waterx(i,j) ! NOTE this is not the entire
01466          tauy = vrel*watery(i,j) ! ocn stress term

01469          cca = umassdtei(i,j) + vrel * cosw      ! alpha, kg/m^2 s

01475              (str(i,j,1) + str(i+1,j,2) + str(i,j+1,3) + str(i+1,j+1,4))
