00263 !  This subroutine attempts to find an optimal (nearly square)
00264 !  2d processor decomposition for a given number of processors.
00265 !
00266 ! !REVISION HISTORY:
00267 !  same as module

00281       nprocs_x, nprocs_y           ! number of procs in each dimension

00283 !EOP
00284 !BOC

00286 !
00287 !  local variables
00288 !
00289 !----------------------------------------------------------------------

00298 !
00299 !  start with an initial guess

00305    nprocs_y = 0

00308       iguess = nint(square)
00309       jguess = nprocs/iguess
00310    elseif (processor_shape == 'square-ice') then ! better for bipolar ice
00311       jguess = nint(square)

00313    elseif (processor_shape == 'slenderX1') then ! 1 proc in y direction
00314       jguess = 1
00315       iguess = nprocs/jguess
00316    else                                  ! 2 processors in y direction
00317       jguess = min(2, nprocs)
00318       iguess = nprocs/jguess
00319    endif
00320 
00321 !----------------------------------------------------------------------
00322 !
00323 !  try various decompositions to find the best
00324 !
00325 !----------------------------------------------------------------------
00326 
00327    proc_loop: do
00328    if (processor_shape == 'square-pop') then
00329       jguess = nprocs/iguess
00330    else
00331       iguess = nprocs/jguess
00332    endif
00333 
00334       if (iguess*jguess == nprocs) then ! valid decomp
00335 
00336          !*** if the blocks can be evenly distributed, it is a
00337          !*** good decomposition
00338          if (mod(nblocks_x,iguess) == 0 .and. &
00339              mod(nblocks_y,jguess) == 0) then
00340             nprocs_x = iguess
00341             nprocs_y = jguess
00342             exit proc_loop
00343 
00344          !*** if the blocks can be evenly distributed in a
00345          !*** transposed direction, it is a good decomposition
00346          else if (mod(nblocks_x,jguess) == 0 .and. &
00347                 mod(nblocks_y,iguess) == 0) then
00348             nprocs_x = jguess
00349             nprocs_y = iguess
00350             exit proc_loop
00351 
00352          !*** A valid decomposition, but keep searching for
00353          !***  a better one
00354          else
00355             if (nprocs_x == 0) then
00356                nprocs_x = iguess
00357                nprocs_y = jguess
00358             endif
00359             if (processor_shape == 'square-pop') then
00360                iguess = iguess - 1
00361                if (iguess == 0) then
00362                   exit proc_loop
00363                else
00364                   cycle proc_loop
00365                endif
00366             else
00367                jguess = jguess - 1
00368                if (jguess == 0) then
00369                   exit proc_loop
00370                else
00371                   cycle proc_loop
00372                endif
00373             endif
00374          endif
00375 
00376       else ! invalid decomp - keep trying
00377 
00378          if (processor_shape == 'square-pop') then
00379             iguess = iguess - 1
00380             if (iguess == 0) then
00381                exit proc_loop
00382             else
00383                cycle proc_loop
00384             endif
00385          else
00386             jguess = jguess - 1
00387             if (jguess == 0) then
00388                exit proc_loop
00389             else
00390                cycle proc_loop
00391             endif
00392          endif
00393       endif
